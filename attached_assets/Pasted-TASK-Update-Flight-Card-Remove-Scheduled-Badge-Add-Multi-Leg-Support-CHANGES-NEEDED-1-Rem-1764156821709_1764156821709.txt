```tsx
// State for multi-leg
const [isMultiLeg, setIsMultiLeg] = useState(false);
const [legs, setLegs] = useState<FlightLeg[]>([]);

// Toggle for multi-leg
<div className="mb-4">
  <label className="flex items-center gap-2 cursor-pointer">
    <input
      type="checkbox"
      checked={isMultiLeg}
      onChange={(e) => {
        setIsMultiLeg(e.target.checked);
        if (e.target.checked && legs.length === 0) {
          // Initialize with 2 legs
          setLegs([
            { departureAirport: '', departureTime: '', arrivalAirport: '', arrivalTime: '', layoverDuration: '' },
            { departureAirport: '', departureTime: '', arrivalAirport: '', arrivalTime: '' }
          ]);
        }
      }}
      className="w-4 h-4"
    />
    <span className="text-sm font-medium">This is a connecting/multi-stop flight</span>
  </label>
</div>

// If multi-leg, show leg inputs
{isMultiLeg && (
  <div className="space-y-6 border-l-4 border-black pl-4">
    {legs.map((leg, index) => (
      <div key={index} className="bg-gray-50 p-4 rounded-lg">
        <div className="flex justify-between items-center mb-3">
          <h4 className="font-semibold">Flight Leg {index + 1}</h4>
          {legs.length > 2 && (
            <button
              type="button"
              onClick={() => setLegs(legs.filter((_, i) => i !== index))}
              className="text-red-500 text-sm hover:underline"
            >
              Remove
            </button>
          )}
        </div>
        
        <div className="grid grid-cols-2 gap-4">

              value={leg.departureAirport}
```

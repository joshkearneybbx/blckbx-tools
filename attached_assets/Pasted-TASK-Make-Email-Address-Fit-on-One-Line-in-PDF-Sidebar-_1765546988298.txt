## ALTERNATIVE - Dynamic Font Sizing:
```tsx
// Calculate font size based on email length
const getEmailFontSize = (email: string) => {
  if (email.length <= 20) return 9;
  if (email.length <= 25) return 8;
  if (email.length <= 30) return 7;
  return 6;
};

// In JSX
<Text style={{
  ...styles.sidebarEmail,
  fontSize: getEmailFontSize(consultant.email)
}}>
  {consultant.email}
</Text>
```

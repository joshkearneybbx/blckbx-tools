```tsx
function SortableTravelItem({ travel, index }: { travel: any; index: number }) {
  const {
    attributes,
    listeners,
    setNodeRef,
    transform,

    isDragging,

            <circle cx="4" cy="12" r="1.5" fill="currentColor"/>

          </svg>
        </button>

        <h3 className="font-semibold text-xl">

        {/* Travel content */}

        <div><strong>Date:</strong> {travel.date}</div>
        {/* ... etc */}
      </div>
    </div>
  );

// In ViewItinerary
const [items, setItems] = useState(itinerary.additionalTravel || []);

const sensors = useSensors(
  useSensor(PointerSensor, {
    activationConstraint: {
      distance: 5,
    },
  })
);

function handleDragEnd(event) {
  const { active, over } = event;
  if (over && active.id !== over.id) {
    setItems((items) => {
      const oldIndex = parseInt(active.id);
      const newIndex = parseInt(over.id);
      return arrayMove(items, oldIndex, newIndex);
    });
  }
}

// JSX
<DndContext sensors={sensors} collisionDetection={closestCenter} onDragEnd={handleDragEnd}>
  <SortableContext items={items.map((_, i) => i.toString())} strategy={verticalListSortingStrategy}>
    <div className="space-y-4">
      {items.map((travel, index) => (
        <SortableTravelItem key={index} travel={travel} index={index} />
      ))}
    </div>
  </SortableContext>
</DndContext>
```

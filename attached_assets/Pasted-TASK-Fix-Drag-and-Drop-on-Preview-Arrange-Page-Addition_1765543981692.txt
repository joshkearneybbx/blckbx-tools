6. VERIFY the SortableItem component has proper props:
```tsx
function SortableItem({ id, type, index, children }) {
  const {
    attributes,
    listeners,
    setNodeRef,
    transform,
    transition,
    isDragging,
  } = useSortable({ id });

  const style = {
    transform: CSS.Transform.toString(transform),
    transition,
    opacity: isDragging ? 0.5 : 1,
  };

  return (
    <div ref={setNodeRef} style={style} className="bg-gray-100 rounded-lg p-6 mb-4">
      {/* Clone children and inject drag handle props */}
      {React.Children.map(children, (child) => {
        if (React.isValidElement(child)) {
          return React.cloneElement(child, { listeners, attributes });
        }
        return child;
      })}
    </div>
  );
}
```

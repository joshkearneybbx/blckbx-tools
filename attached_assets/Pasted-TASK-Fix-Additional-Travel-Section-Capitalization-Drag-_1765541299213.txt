2. UPDATE: Add actual drag-and-drop functionality (optional but recommended)

If you want functional drag-and-drop reordering, install react-beautiful-dnd:
```bash
npm install react-beautiful-dnd
npm install --save-dev @types/react-beautiful-dnd
```

Then update the Additional Travel section:
```tsx
import { DragDropContext, Droppable, Draggable } from 'react-beautiful-dnd';

// Add state for reordering
const [additionalTravelItems, setAdditionalTravelItems] = useState(itinerary.additionalTravel || []);

// Handle drag end
const handleDragEnd = (result) => {
  if (!result.destination) return;
  
  const items = Array.from(additionalTravelItems);
  const [reorderedItem] = items.splice(result.source.index, 1);
  items.splice(result.destination.index, 0, reorderedItem);
  
  setAdditionalTravelItems(items);
  
  // Optional: Save to database
  // updateItineraryOrder(itinerary.id, items);
};

// In the JSX:
<DragDropContext onDragEnd={handleDragEnd}>
  <Droppable droppableId="additional-travel">
    {(provided) => (
      <div 
        {...provided.droppableProps} 
        ref={provided.innerRef}
        className="space-y-4"
      >
        {additionalTravelItems.map((travel, index) => (
          <Draggable 
            key={`travel-${index}`} 
            draggableId={`travel-${index}`} 
            index={index}
          >
            {(provided, snapshot) => (
              <div
                ref={provided.innerRef}
                {...provided.draggableProps}
                className={`bg-gray-100 rounded-lg shadow-sm border border-gray-200 overflow-hidden ${
                  snapshot.isDragging ? 'shadow-xl ring-2 ring-black' : ''
                }`}
              >
                {/* Header with drag handle */}
                <div className="bg-white px-6 py-3 flex items-center justify-between border-b border-gray-200">
                  <div className="flex items-center gap-3">
                    {/* Drag handle - use provided.dragHandleProps */}
                    <div 
                      {...provided.dragHandleProps}
                      className="flex flex-col gap-1 cursor-move hover:opacity-60 transition-opacity"
                    >
                      <div className="flex gap-1">
                        <div className="w-1.5 h-1.5 rounded-full bg-gray-400"></div>
                        <div className="w-1.5 h-1.5 rounded-full bg-gray-400"></div>
                      </div>
                      <div className="flex gap-1">
                        <div className="w-1.5 h-1.5 rounded-full bg-gray-400"></div>
                        <div className="w-1.5 h-1.5 rounded-full bg-gray-400"></div>
                      </div>
                    </div>
                    {/* Capitalize travel type */}
                    <h3 className="font-semibold text-lg">
                      {travel.type.charAt(0).toUpperCase() + travel.type.slice(1)}
                    </h3>

                </div>
              </div>
            )}
          </Draggable>
        ))}
        {provided.placeholder}
      </div>
    )}

</DragDropContext>
```
